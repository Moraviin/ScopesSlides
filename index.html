What is it scope ?


JavaScript Function Scope

In JavaScript there are two types of scope:

  • Global scope
  • Local scope

Local scope


function myFunction() {
  var scope = 'local';
  console.log(scope); // local
}
console.log(scope); // ERROR

Global scope


var scope = 'global';
function myFunction() {
  console.log(scope); // global
}
console.log(scope); // global

Automatically Global

If you assign a value to a variable that has not been declared, it will automatically become a GLOBAL variable. 


myFunction();
console.log(scope); // hi
function myFunction() {
  scope = 'hi';
}

Global scope in HTML


var awesomeVariable = 'global';
console.log(window.awesomeVariable);// global

Nested variables


function myFunction() {
  var welcome = ' O, hi';
  function nestedFunction() {
    var user = ' Mark';
    console.log(welcome + user);
  }
  nestedFunction(); // O, hi Mark
}

Shadowing function


function myFunction() {
  var welcome = ' O, hi';
  function nestedFunction() {
	var welcome = 'Welcome';
	var user = ' Mark';
	console.log(welcome + user);
  }
  nestedFunction(); // Welcome Mark
}

Hoisting

Hoisting is JavaScript's default behavior of moving all declarations to the top of the current scope. 



console.log('something unnecessary');
var up = 'value';
		  

var up;
console.log('something unnecessary');
up = 'value';

Hoisting + Shadowing


function myFunction() {
  var welcome = ' O, hi';
  function nestedFunction() {
    var user = ' Mark';
    console.log(welcome + user);
    var welcome = "You're welcome" 
  }
  nestedFunction(); // undefined Mark
}

Be carefull

let and const are NOT hoisted

Let and Const

ES2015 introduced two important new JavaScript keywords: let and const.

These two keywords provide Block Scope variables (and constants) in JavaScript.

Block Scope


if(condition){
  let x = 'value';
  console.log(x);// value 
  //Block scope
} 
console.log(x)//ERROR

const vs let

let = var + block scope

const = let + constant


  let x = 'value';
  x = 0;
  x++;
  
  const road = 'any';
  road = 'not any'; // ERROR
  const arr = [];
  arr.push('something');
  arr = ['something', 1]; // ERROR
